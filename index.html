// ============================================================ // Budget Tracker - Google Apps Script Backend // ============================================================ const SHEET_TRANSACTIONS = "Transactions"; const SHEET_SETTINGS = "Settings"; const SHEET_CONSTANTS = "Constants"; const SHEET_INCOMES = "Incomes"; const SHEET_CHANGELOG = "Changelog"; const SECRET_KEY = "Duek2026"; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { let data = {}; try { if (e.parameter && Object.keys(e.parameter).length > 0) { data = e.parameter; } else if (e.postData) { data = JSON.parse(e.postData.contents); } if (data.secret !== SECRET_KEY) return respond({ error: "Unauthorized" }, data.callback); let result; switch (data.action) { case "addTransaction": result = addTransaction(data); break; case "getTransactions": result = getTransactions(data); break; case "deleteTransaction": result = deleteTransaction(data); break; case "updateTransaction": result = updateTransaction(data); break; case "getSettings": result = getSettings(); break; case "saveSettings": result = saveSettings(data); break; case "getConstants": result = getConstants(data); break; case "saveConstants": result = saveConstants(data); break; case "getIncomes": result = getIncomes(data); break; case "saveIncomes": result = saveIncomes(data); break; case "getYearlySummary": result = getYearlySummary(data); break; default: result = { error: "Unknown action: " + data.action }; } return respond(result, data.callback); } catch (err) { return respond({ error: err.toString() }, data ? data.callback : null); } } function respond(data, callback) { if (callback) { const out = ContentService.createTextOutput(callback + '(' + JSON.stringify(data) + ')'); out.setMimeType(ContentService.MimeType.JAVASCRIPT); return out; } const out = ContentService.createTextOutput(JSON.stringify(data)); out.setMimeType(ContentService.MimeType.JSON); return out; } // ============================================================ // HELPERS // ============================================================ function normalizeDate(raw) { if (raw instanceof Date) { return raw.getFullYear() + '-' + String(raw.getMonth()+1).padStart(2,'0') + '-' + String(raw.getDate()).padStart(2,'0'); } return String(raw); } // Normalize month to YYYYMM (6-digit number string, e.g. "202603") // Sheets cannot auto-convert a 6-digit number to a date function normalizeMonth(raw) { if (!raw) return ''; if (raw instanceof Date) { return raw.getFullYear() + String(raw.getMonth()+1).padStart(2,'0'); } const s = String(raw).trim(); if (/^\d{6}$/.test(s)) return s; // already YYYYMM if (/^\d{4}-\d{2}$/.test(s)) return s.replace('-',''); // YYYY-MM → YYYYMM const parts = s.split('/'); if (parts.length === 3) { // D/M/YYYY (Israel Sheets locale) return parts[2].trim() + parts[1].trim().padStart(2,'0'); } if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s.substring(0,4) + s.substring(5,7); // YYYY-MM-DD return s; } function monthLabel(yyyymm) { // "202603" → "March 2026" if (!yyyymm || yyyymm.length < 6) return yyyymm; const y = parseInt(yyyymm.substring(0,4)); const m = parseInt(yyyymm.substring(4,6)) - 1; return new Date(y, m, 1).toLocaleString('default', { month: 'long', year: 'numeric' }); } function getOrCreateSheet(name, headers) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1,1,1,headers.length).setFontWeight("bold"); } return sheet; } // ============================================================ // TRANSACTIONS // ============================================================ function addTransaction(data) { const sheet = getOrCreateSheet(SHEET_TRANSACTIONS, ["ID","Date","Type","Amount","Description","Person","IncomeSource","CreatedAt"]); const id = Utilities.getUuid(); sheet.appendRow([ id, data.date, data.type, parseFloat(data.amount), data.description || "", data.person || "", data.incomeSource || "", new Date().toISOString() ]); return { success: true, id: id }; } function getTransactions(data) { const sheet = getOrCreateSheet(SHEET_TRANSACTIONS, ["ID","Date","Type","Amount","Description","Person","IncomeSource","CreatedAt"]); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { transactions: [] }; let transactions = rows.slice(1).map(row => ({ id: String(row[0]), date: normalizeDate(row[1]), type: String(row[2]), amount: parseFloat(row[3]) || 0, description: String(row[4]), person: String(row[5] || ''), incomeSource: String(row[6] || ''), // col 6 = IncomeSource (new); old rows return '' createdAt: String(row[7] || row[6] || '') // col 7 = CreatedAt; fallback to col 6 for old rows })); if (data.month) transactions = transactions.filter(t => t.date.startsWith(data.month)); if (data.year) transactions = transactions.filter(t => t.date.startsWith(data.year)); return { transactions }; } function deleteTransaction(data) { const sheet = getOrCreateSheet(SHEET_TRANSACTIONS, ["ID","Date","Type","Amount","Description","Person","IncomeSource","CreatedAt"]); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i+1); return { success: true }; } } return { error: "Not found" }; } function updateTransaction(data) { const sheet = getOrCreateSheet(SHEET_TRANSACTIONS, ["ID","Date","Type","Amount","Description","Person","IncomeSource","CreatedAt"]); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.getRange(i+1,2).setValue(data.date); sheet.getRange(i+1,3).setValue(data.type); sheet.getRange(i+1,4).setValue(parseFloat(data.amount)); sheet.getRange(i+1,5).setValue(data.description || ""); return { success: true }; } } return { error: "Not found" }; } // ============================================================ // SETTINGS // ============================================================ function getSettings() { const sheet = getOrCreateSheet(SHEET_SETTINGS, ["Key","Value"]); const rows = sheet.getDataRange().getValues(); const settings = {}; rows.slice(1).forEach(row => { settings[String(row[0])] = row[1]; }); return { settings }; } function saveSettings(data) { const sheet = getOrCreateSheet(SHEET_SETTINGS, ["Key","Value"]); const rows = sheet.getDataRange().getValues(); const toSave = { budget: data.budget }; Object.entries(toSave).forEach(([key, value]) => { if (value === undefined || value === null) return; let found = false; for (let i = 1; i < rows.length; i++) { if (rows[i][0] === key) { sheet.getRange(i+1,2).setValue(value); rows[i][1] = value; found = true; break; } } if (!found) { sheet.appendRow([key, value]); rows.push([key, value]); } }); return { success: true }; } // ============================================================ // CONSTANTS — upsert per name+month, never delete old data // ============================================================ function getConstants(data) { const sheet = getOrCreateSheet(SHEET_CONSTANTS, ["Name","Amount","Month"]); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { constants: [], month: null }; const requestedMonth = normalizeMonth(data.month || ''); const allMonths = [...new Set( rows.slice(1).map(r => normalizeMonth(r[2])).filter(Boolean) )].sort().reverse(); const useMonth = allMonths.find(m => m === requestedMonth) || allMonths.find(m => m < requestedMonth) || allMonths[0] || null; if (!useMonth) return { constants: [], month: null }; // Take the LATEST entry per name for this month (last row wins) const latestByName = {}; rows.slice(1).forEach(r => { if (normalizeMonth(r[2]) === useMonth) { latestByName[String(r[0]).trim()] = parseFloat(r[1]) || 0; } }); const constants = Object.entries(latestByName).map(([name, amount]) => ({ name, amount })); return { constants, month: useMonth }; } function saveConstants(data) { const sheet = getOrCreateSheet(SHEET_CONSTANTS, ["Name","Amount","Month"]); const month = normalizeMonth(data.month); let constants = data.constants; if (typeof constants === 'string') constants = JSON.parse(constants); if (!Array.isArray(constants)) return { error: "Invalid constants" }; // Read existing rows — find which names already exist for this month const rows = sheet.getDataRange().getValues(); const existingByName = {}; // name → row index (1-based) rows.slice(1).forEach((r, i) => { if (normalizeMonth(r[2]) === month) { existingByName[String(r[0]).trim()] = i + 2; // +2: header + 0-index } }); // Upsert: update existing rows in place, append new ones constants.forEach(c => { const name = c.name.trim(); const amount = parseFloat(c.amount) || 0; if (existingByName[name] !== undefined) { sheet.getRange(existingByName[name], 2).setValue(amount); // update amount only } else { sheet.appendRow([name, amount, month]); } }); // Remove rows for this month that are no longer in the list (user deleted them) const newNames = new Set(constants.map(c => c.name.trim())); const rowsAfter = sheet.getDataRange().getValues(); for (let i = rowsAfter.length - 1; i >= 1; i--) { if (normalizeMonth(rowsAfter[i][2]) === month && !newNames.has(String(rowsAfter[i][0]).trim())) { sheet.deleteRow(i + 1); } } formatMonthlySheet(sheet); appendChangelog('constants', month, constants); return { success: true }; } // ============================================================ // INCOMES — same upsert logic // ============================================================ function getIncomes(data) { const sheet = getOrCreateSheet(SHEET_INCOMES, ["Name","Amount","Month"]); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { incomes: [], month: null }; const requestedMonth = normalizeMonth(data.month || ''); const allMonths = [...new Set( rows.slice(1).map(r => normalizeMonth(r[2])).filter(Boolean) )].sort().reverse(); const useMonth = allMonths.find(m => m === requestedMonth) || allMonths.find(m => m < requestedMonth) || allMonths[0] || null; if (!useMonth) return { incomes: [], month: null }; const latestByName = {}; rows.slice(1).forEach(r => { if (normalizeMonth(r[2]) === useMonth) { latestByName[String(r[0]).trim()] = parseFloat(r[1]) || 0; } }); const incomes = Object.entries(latestByName).map(([name, amount]) => ({ name, amount })); return { incomes, month: useMonth }; } function saveIncomes(data) { const sheet = getOrCreateSheet(SHEET_INCOMES, ["Name","Amount","Month"]); const month = normalizeMonth(data.month); let incomes = data.incomes; if (typeof incomes === 'string') incomes = JSON.parse(incomes); if (!Array.isArray(incomes)) return { error: "Invalid incomes" }; const rows = sheet.getDataRange().getValues(); const existingByName = {}; rows.slice(1).forEach((r, i) => { if (normalizeMonth(r[2]) === month) { existingByName[String(r[0]).trim()] = i + 2; } }); incomes.forEach(inc => { const name = inc.name.trim(); const amount = parseFloat(inc.amount) || 0; if (existingByName[name] !== undefined) { sheet.getRange(existingByName[name], 2).setValue(amount); } else { sheet.appendRow([name, amount, month]); } }); const newNames = new Set(incomes.map(i => i.name.trim())); const rowsAfter = sheet.getDataRange().getValues(); for (let i = rowsAfter.length - 1; i >= 1; i--) { if (normalizeMonth(rowsAfter[i][2]) === month && !newNames.has(String(rowsAfter[i][0]).trim())) { sheet.deleteRow(i + 1); } } formatMonthlySheet(sheet); appendChangelog('incomes', month, incomes); return { success: true }; } // ============================================================ // SHEET FORMATTING — month name + total row per month // ============================================================ function formatMonthlySheet(sheet) { try { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return; // Group rows by normalized month const monthGroups = {}; rows.slice(1).forEach((r, i) => { const m = normalizeMonth(r[2]); if (!monthGroups[m]) monthGroups[m] = []; monthGroups[m].push({ rowIdx: i + 2, data: r }); // rowIdx is 1-based sheet row }); // Color alternating months const months = Object.keys(monthGroups).sort(); const colors = ['#1a2030', '#1e2a3a']; // subtle alternating dark colors months.forEach((m, mi) => { const bg = colors[mi % 2]; monthGroups[m].forEach(({ rowIdx }) => { sheet.getRange(rowIdx, 1, 1, 3).setBackground(bg); }); }); // Set month name in column C (visible label for humans) // Already stored as YYYYMM, no change needed to data — just display // Add a note in the header if not already there const headerNote = sheet.getRange(1, 3).getNote(); if (!headerNote) sheet.getRange(1, 3).setNote('Format: YYYYMM (e.g. 202603 = March 2026)'); } catch(e) { Logger.log('formatMonthlySheet error: ' + e); } } // ============================================================ // CHANGELOG — append every save for audit trail // ============================================================ function appendChangelog(type, month, items) { try { const sheet = getOrCreateSheet(SHEET_CHANGELOG, ["Timestamp","Type","Month","Name","Amount"]); const ts = new Date().toISOString(); items.forEach(item => { sheet.appendRow([ts, type, month, item.name, parseFloat(item.amount)||0]); }); } catch(e) { Logger.log('Changelog error: ' + e); } } // ============================================================ // YEARLY SUMMARY // ============================================================ function getYearlySummary(data) { const year = data.year || String(new Date().getFullYear()); const constSheet = getOrCreateSheet(SHEET_CONSTANTS, ["Name","Amount","Month"]); const incSheet = getOrCreateSheet(SHEET_INCOMES, ["Name","Amount","Month"]); const constRows = constSheet.getDataRange().getValues().slice(1); const incRows = incSheet.getDataRange().getValues().slice(1); const monthlyConstants = {}; const monthlyIncomes = {}; constRows.forEach(r => { const m = normalizeMonth(r[2]); if (m.startsWith(year)) { monthlyConstants[m] = (monthlyConstants[m]||0) + (parseFloat(r[1])||0); } }); incRows.forEach(r => { const m = normalizeMonth(r[2]); if (m.startsWith(year)) { monthlyIncomes[m] = (monthlyIncomes[m]||0) + (parseFloat(r[1])||0); } }); return { monthlyConstants, monthlyIncomes }; } // ============================================================ // ONE-TIME DEDUP CLEANUP (run manually from Apps Script editor) // ============================================================ function deduplicateSheets() { deduplicateSheet(SHEET_CONSTANTS); deduplicateSheet(SHEET_INCOMES); Logger.log('Done.'); } function deduplicateSheet(sheetName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) return; const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return; const seen = {}; rows.slice(1).forEach(r => { const name = String(r[0]).trim(); const month = normalizeMonth(r[2]); if (!name || !month) return; seen[name + '|' + month] = [name, parseFloat(r[1]) || 0, month]; }); sheet.clearContents(); sheet.appendRow(["Name","Amount","Month"]); sheet.getRange(1,1,1,3).setFontWeight("bold"); Object.values(seen) .sort((a,b) => a[2].localeCompare(b[2]) || a[0].localeCompare(b[0])) .forEach(r => sheet.appendRow(r)); Logger.log(sheetName + ': ' + Object.keys(seen).length + ' unique rows'); }
